const readline = require("readline");
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

const input = [];
rl.on("line", function (line) {
  input.push(line);
}).on("close", function () {
  solution(input);
  process.exit();
});

function solution(input) {
  const n = +input[0];
  let ans = 666;
  let cnt = 1;

  while (cnt !== n) {
    ++ans;
    if (String(ans).includes("666")) ++cnt;
  }
  console.log(ans);
}
