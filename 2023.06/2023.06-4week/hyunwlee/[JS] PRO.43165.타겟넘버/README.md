let ans = 0;
function solution(numbers, target) {
  recurse(0, 0, target, numbers);

function recurse(depth, cur, target, numbers) {
  if (depth === numbers.length) {
    if (cur === target)
      ++ans;
    return ;
  }
  recurse(depth + 1, cur - numbers[depth], target, numbers);
  recurse(depth + 1, cur + numbers[depth], target, numbers);
}
