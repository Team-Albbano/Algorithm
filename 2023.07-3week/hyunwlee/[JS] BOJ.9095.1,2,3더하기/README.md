// d[n] = d[n - 1] + 1 + d[n - 2] + 1 d[n - 3] + 1;
function topDown(n, memo) {
  if (memo[n] > 0)
    return memo[n];
  if (n === 0)
    return (0);
  if (n === 1)
    return (memo[1] = 1);
  if (n === 2)
    return (memo[2] = 2);
  if (n === 3)
    return (memo[3] = 4);
  memo[n] = topDown((n - 1), memo) + topDown((n - 2), memo) + topDown((n - 3), memo);
  return memo[n];
}
